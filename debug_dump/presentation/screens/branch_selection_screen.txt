   9: 
  10: class BranchSelectionScreen extends ConsumerWidget {

  12: 
  13:   @override
  14:   Widget build(BuildContext context, WidgetRef ref) {
  15:     final branchesAsync = ref.watch(branchesProvider);
  16: 
  17:     return Scaffold(
  18:       appBar: CustomAppBar(
  19:         title: AppStrings.selectBranch,
  20:         showBackButton: true,
  21:       ),
  22:       body: branchesAsync.when(
  23:         data: (branches) {
  24:           if (branches.isEmpty) {
  25:             return Center(
  26:               child: Text(AppStrings.noBranchesAvailable),
  27:             );
  28:           }
  29: 
  30:           return ListView.builder(
  31:             padding: const EdgeInsets.all(AppDimensions.screenPaddingHorizontal),
  32:             itemCount: branches.length,
  33:             itemBuilder: (context, index) {
  34:               final branch = branches[index];
  35:               return Padding(
  36:                 padding: const EdgeInsets.only(bottom: AppDimensions.paddingMedium),
  37:                 child: BranchCard(
  38:                   branch: branch,
  39:                   onTap: () {
  40:                     ref.read(selectedBranchProvider.notifier).state = branch;
  41:                     Navigator.of(context).push(
  42:                       MaterialPageRoute(
  43:                         builder: (context) => const ServiceSelectionScreen(),
  44:                       ),
  45:                     );
  46:                   },
  47:                 ),
  48:               );
  49:             },
  50:           );
  51:         },
  52:         loading: () => const Center(child: CircularProgressIndicator()),
  53:         error: (error, stackTrace) => Center(
  54:           child: Text('Erro: $error'),
  55:         ),
  56:       ),
  57:     );
  58:   }
  59: }
